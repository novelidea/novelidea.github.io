public class moveForwardAndBackward : MonoBehaviour {

    Vector3 origin;
    float distance = 6;
    float speed = 0.3f;

    void Start(){
        origin = transform.position;
    }

    void Update () {
        transform.position = origin + (transform.forward*(Mathf.PingPong(Time.time*distance*speed + (distance*speed/2), distance*2)-distance));
    }

}
