有向图的单源最短路径算法实现

算法

有向图的实现

首先，我们再来讨论一下图。图是数据结构中相当重要的一部分，与之相关的算法也有很多，但是如果要让你代码实现有向图，相信很多人都会写很长的时间。下面就用Java实现一个Adjacency List存储的有向图。关于图的存储，可以参见这篇博客。

这里主要定义了三个类，点(Vertex)、边(Edge)、有向图（Digraph）。

package GraphImplementation;

/**
 * Created by novelidea on 2/28/16.
 */
public class Vertex {
    private String id;
    private int distance = Integer.MAX_VALUE;
    public Vertex(String id){
        this.id = id;
    }
    public String getId(){
        return id;
    }
    public int getDistance(){
        return distance;
    }
    public void setDistance(int dis){
        distance = dis;
    }
}

package GraphImplementation;

/**
 * Created by novelidea on 2/28/16.
 */
public class Edge {
    private Vertex in;
    private Vertex out;
    private int weight;

    Edge(Vertex i, Vertex o, int w){
        in = i;
        out = o;
        weight = w;
    }
    public Vertex getIn(){
        return in;
    }
    public Vertex getOut(){
        return out;
    }
    public int getWeight(){
        return weight;
    }
    public Vertex getAdjacentVertex(Vertex v) {
        return v.getId().equals(in.getId()) ? out : in;
    }
}

package GraphImplementation;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;

/**
 * Created by novelidea on 2/28/16.
 */
public class Digraph {
    private HashMap<Vertex, ArrayList<Edge>> adjacentList = new HashMap<Vertex, ArrayList<Edge>>();
    public void addVertex(Vertex v){
        if(adjacentList.containsKey(v)) return;
        adjacentList.put(v, new ArrayList<>());
    }
    public void addEdge(Vertex in, Vertex out, int weight){
        this.addVertex(in);
        this.addVertex(out);
        adjacentList.get(in).add(new Edge(in, out, weight));
    }
    public int outDegree(Vertex v){
        return adjacentList.get(v).size();
    }
    public HashMap<Vertex, ArrayList<Edge>> getMap(){
        return adjacentList;
    }
    public ArrayList<Edge> getAdjacentEdges(Vertex v){
        return adjacentList.get(v);
    }
}

实现有向图单源最短路径主要是Dijkstra算法，下面为算法实现。

public class Dijkstra {
private Digraph graph;

public Dijkstra(Digraph g){
    if(graph == null){
        throw new NullPointerException("the input graph cannot be null");
    }
    this.graph = g;
}

public class VertexComparator implements Comparator&lt;Vertex&gt; {
    @Override
    public int compare(Vertex v1, Vertex v2){
        return Integer.compare(v1.getDistance(), v2.getDistance());
    }
}

public Set&lt;Vertex&gt; findShortestPath(String sourceID){
    Queue&lt;Vertex&gt; queue = new PriorityQueue&lt;&gt;(10, new VertexComparator());
    for(Map.Entry&lt;Vertex, ArrayList&lt;Edge&gt;&gt; e : graph.getMap().entrySet()){
        Vertex v = e.getKey();
        if(v.getId().equals(sourceID)){
            v.setDistance(0);
            queue.add(v);
        }
    }

    Set&lt;Vertex&gt; visited = new HashSet&lt;Vertex&gt;();

    while (!queue.isEmpty()){
        Vertex src = queue.poll();
        visited.add(src);

        for(Edge e : graph.getAdjacentEdges(src)){
            Vertex v = e.getAdjacentVertex(src);
            if(! visited.contains(v)){
                int newDistance = src.getDistance() + e.getWeight();
                if(newDistance &lt; v.getDistance()){
                    v.setDistance(newDistance);
                    visited.add(v);
                }
            }
        }
    }
    return graph.getMap().keySet();
}

}
