面向对象设计

Feb 28, 2016

本文主要讨论用面相对象的思想设计一副纸牌，并且实现洗牌算法。设计了两个类Card, Deck，以及两个枚举Suit, Face.

package com.pengfeixing.ood;

/**

  • Created by novelidea on 2/28/16.
    */
    public enum Suit {
    Club(0), Diamond(1), Heart(2), Spade(3);
    private int value;
    private Suit(int v){
    value = v;
    }
    public int getValue(){
    return value;
    }
    public static Suit getSuitFromValue(int value){
    switch (value){
    case 0 : return Suit.Club;
    case 1 : return Suit.Diamond;
    case 2 : return Suit.Heart;
    case 3 : return Suit.Spade;
    default: return null;
}
    }
    }

    public enum Face {
    A(1), Two(2), Three(3), Four(4), Five(5), Six(6), Seven(7), Eight(8), Nine(9), Ten(10), Jack(11), Queen(12), King(13);

    private int value;
    private Face(int v){

    value = v;

    }
    public int getValue(){

    return value;

    }
    public static Face getFaceFromValue(int v){

    switch (v){
    case 1 : return Face.A;
    case 2 : return Face.Two;
    case 3 : return Face.Three;
    case 4 : return Face.Four;
    case 5 : return Face.Five;
    case 6 : return Face.Six;
    case 7 : return Face.Seven;
    case 8 : return Face.Eight;
    case 9 : return Face.Nine;
    case 10 : return Face.Ten;
    case 11 : return Face.Jack;
    case 12 : return Face.Queen;
    case 13 : return Face.King;
    default: return null;
}

    }
    }

    package com.pengfeixing.ood;

/**

  • Created by novelidea on 2/28/16.
    */
    public class Card {
    private Face faceValue;
    private Suit suit;
    public Card(Face v, Suit s){
    faceValue = v;
suit = s;
    }
    public Card(Card otherCard){
    faceValue = otherCard.faceValue;
suit = otherCard.suit;
    }
    public Face getFaceValue(){
    return faceValue;
    }
    public Suit getSuit(){
    return suit;
    }
    }

    package com.pengfeixing.ood;

/**

  • Created by novelidea on 2/28/16.
    */
    import java.util.ArrayList;
    import java.util.Random;

public class Deck {
private static int SUIT_PER_DECK = 4;
private static int CARDS_PER_SUIT = 13;
private ArrayList<Card> cards;
public Deck(){
cards = new ArrayList<Card>(SUIT_PER_DECK CARDS_PER_SUIT);
int index = 0;
for (int i = 0; i < SUIT_PER_DECK; i ++) {
for (int j = 1; j <= CARDS_PER_SUIT; j ++){
index = i CARDS_PER_SUIT + j - 1;
Card card = new Card(Face.getFaceFromValue(j), Suit.getSuitFromValue(i));
cards.add(card);
}
}
}
public void Shuffle(){
if(cards.isEmpty()){
return;
}
Random rand = new Random();
for (int i = SUIT_PER_DECK * CARDS_PER_SUIT - 1; i > 0; i –){
int j = rand.nextInt(i);
Card tmp = new Card(cards.get(i));
cards.set(i, cards.get(j));
cards.set(j, tmp);
}
return;
}

public void printCards(){
    for(int i = 0; i &lt; CARDS_PER_SUIT * SUIT_PER_DECK; i ++){
        Card card = cards.get(i);
        System.out.println(card.getFaceValue() + ",  " + card.getSuit());
    }
}

}

package com.pengfeixing.ood;

public class Main {
public static void main(String[] args) {
Deck deck = new Deck();
deck.Shuffle();
deck.printCards();
}
}
